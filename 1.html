# Функция для удаления выбросов по методу IQR
remove_outliers <- function(x) {
  Q1 <- quantile(x, 0.25, na.rm = TRUE)
  Q3 <- quantile(x, 0.75, na.rm = TRUE)
  IQR <- Q3 - Q1
  lower_bound <- Q1 - 1.5 * IQR
  upper_bound <- Q3 + 1.5 * IQR
  x[x >= lower_bound & x <= upper_bound]
}

# Функция для теста Шапиро-Уилка с ограничением выборки
shapiro_test_safe <- function(data_vector, group_name) {
  n <- length(data_vector)
  
  if(n > 5000) {
    set.seed(123)
    data_sample <- sample(data_vector, 5000)
    shapiro_result <- shapiro.test(data_sample)
    cat("Тест Шапиро-Уилка для", group_name, "(выборка 5000 из", n, "наблюдений):\n")
  } else {
    shapiro_result <- shapiro.test(data_vector)
    cat("Тест Шапиро-Уилка для", group_name, "(все наблюдения, n =", n, "):\n")
  }
  
  print(shapiro_result)
  cat("Нормальность:", ifelse(shapiro_result$p.value > 0.05, "✓ Принимаем", "✗ Отвергаем"), 
      "(p-value =", round(shapiro_result$p.value, 4), ")\n\n")
  
  return(shapiro_result)
}

print("Тест Левена на равенство дисперсий:")
## [1] "Тест Левена на равенство дисперсий:"
print(levene_test)
## Levene's Test for Homogeneity of Variance (center = median)
##         Df F value Pr(>F)
## group    3  0.2367 0.8708
##       1808
# Bartlett test (альтернативный тест)
bartlett_test <- bartlett.test(total_spend ~ city, data = city_clean)
print("Тест Бартлетта на равенство дисперсий:")
## [1] "Тест Бартлетта на равенство дисперсий:"
print(bartlett_test)
## 
##  Bartlett test of homogeneity of variances
## 
## data:  total_spend by city
## Bartlett's K-squared = 1.2481, df = 3, p-value = 0.7415
# Вывод по результатам проверки дисперсий
if(levene_test$`Pr(>F)`[1] > 0.05) {
  cat("✓ Дисперсии можно считать равными (p-value =", round(levene_test$`Pr(>F)`[1], 4), ")\n")
} else {
  cat("✗ Дисперсии статистически значимо различаются (p-value =", round(levene_test$`Pr(>F)`[1], 4), ")\n")
}
## ✓ Дисперсии можно считать равными (p-value = 0.8708 )
## 3. Проведение ANOVA теста (даже если предположения нарушены)
cat("\n=== ОДНОФАКТОРНЫЙ ANOVA ТЕСТ ===\n\n")
## 
## === ОДНОФАКТОРНЫЙ ANOVA ТЕСТ ===
# Однофакторный ANOVA
anova_result <- aov(total_spend ~ city, data = city_clean)
summary_anova <- summary(anova_result)
print("Результаты ANOVA:")
## [1] "Результаты ANOVA:"
print(summary_anova)
##               Df    Sum Sq   Mean Sq F value Pr(>F)
## city           3 9.187e+09 3.062e+09   1.246  0.292
## Residuals   1808 4.444e+12 2.458e+09
# Непараметрическая альтернатива (Крускал-Уоллис) если нормальность нарушена
cat("\n=== НЕПАРАМЕТРИЧЕСКИЙ ТЕСТ КРУСКАЛА-УОЛЛИСА (альтернатива) ===\n\n")
## 
## === НЕПАРАМЕТРИЧЕСКИЙ ТЕСТ КРУСКАЛА-УОЛЛИСА (альтернатива) ===
kruskal_test <- kruskal.test(total_spend ~ city, data = city_clean)
print("Тест Крускала-Уоллиса:")
## [1] "Тест Крускала-Уоллиса:"
print(kruskal_test)
## 
##  Kruskal-Wallis rank sum test
## 
## data:  total_spend by city
## Kruskal-Wallis chi-squared = 8.0742, df = 3, p-value = 0.0445
# Интерпретация результатов
p_value_anova <- summary_anova[[1]]$`Pr(>F)`[1]
p_value_kruskal <- kruskal_test$p.value

cat("\n=== ИНТЕРПРЕТАЦИЯ РЕЗУЛЬТАТОВ ===\n\n")
## 
## === ИНТЕРПРЕТАЦИЯ РЕЗУЛЬТАТОВ ===
# Проверяем, выполнены ли предположения ANOVA
normality_ok <- all(sapply(shapiro_results, function(x) x$p.value > 0.05))
variance_ok <- levene_test$`Pr(>F)`[1] > 0.05

cat("Предположения ANOVA:\n")
## Предположения ANOVA:
cat("- Нормальность распределения:", ifelse(normality_ok, "✓ выполнено", "✗ нарушено"), "\n")
## - Нормальность распределения: ✗ нарушено
cat("- Равенство дисперсий:", ifelse(variance_ok, "✓ выполнено", "✗ нарушено"), "\n\n")
## - Равенство дисперсий: ✓ выполнено
if(normality_ok && variance_ok) {
  cat("ИСПОЛЬЗУЕМ ANOVA РЕЗУЛЬТАТ:\n")
  if(p_value_anova < 0.05) {
    cat("✓ Отвергаем нулевую гипотезу H0\n")
    cat("Статистически значимые различия в средних тратах между городами обнаружены (p-value =", 
        round(p_value_anova, 4), ")\n")
  } else {
    cat("✗ Не отвергаем нулевую гипотезу H0\n")
    cat("Статистически значимых различий в средних тратах между городами не обнаружено (p-value =", 
        round(p_value_anova, 4), ")\n")
  }
} else {
  cat("ИСПОЛЬЗУЕМ ТЕСТ КРУСКАЛА-УОЛЛИСА (предположения ANOVA нарушены):\n")
  if(p_value_kruskal < 0.05) {
    cat("✓ Отвергаем нулевую гипотезу H0\n")
    cat("Статистически значимые различия в распределении трат между городами обнаружены (p-value =", 
        round(p_value_kruskal, 4), ")\n")
  } else {
    cat("✗ Не отвергаем нулевую гипотезу H0\n")
    cat("Статистически значимых различий в распределении трат между городами не обнаружено (p-value =", 
        round(p_value_kruskal, 4), ")\n")
  }
}
## ИСПОЛЬЗУЕМ ТЕСТ КРУСКАЛА-УОЛЛИСА (предположения ANOVA нарушены):
## ✓ Отвергаем нулевую гипотезу H0
## Статистически значимые различия в распределении трат между городами обнаружены (p-value = 0.0445 )
## 4. Пост-хок анализ (если тесты значимы)
if(p_value_anova < 0.05 || p_value_kruskal < 0.05) {
  cat("\n=== POST-HOC АНАЛИЗ ===\n\n")
  
  if(normality_ok && variance_ok) {
    # Tukey HSD test для ANOVA
    tukey_result <- TukeyHSD(anova_result)
    print("Попарные сравнения по методу Тьюки (ANOVA):")
    print(tukey_result)
    
    # Визуализация доверительных интервалов
    par(mar = c(5, 10, 4, 2)) # Увеличиваем отступ слева для подписей
    plot(tukey_result, las = 1)
    title("Попарные сравнения городов (метод Тьюки)")
    
  } else {
    # Dunn test для Крускала-Уоллиса
    if(require(dunn.test, quietly = TRUE)) {
      dunn_result <- dunn.test(city_clean$total_spend, city_clean$city, method = "bonferroni")
      cat("Попарные сравнения по методу Данна (Крускал-Уоллис):\n")
      print(dunn_result)
    } else {
      cat("Для post-hoc анализа Крускала-Уоллиса необходимо установить пакет 'dunn.test'\n")
      cat("Установите пакет: install.packages('dunn.test')\n")
      
      # Альтернатива: попарные тесты Уилкоксона с поправкой Бонферрони
      cat("\nАльтернатива: попарные тесты Уилкоксона с поправкой Бонферрони:\n")
      city_pairs <- combn(unique(city_clean$city), 2, simplify = FALSE)
      
      for(pair in city_pairs) {
        city1_data <- city_clean %>% filter(city == pair[1]) %>% pull(total_spend)
        city2_data <- city_clean %>% filter(city == pair[2]) %>% pull(total_spend)
        
        wilcox_test <- wilcox.test(city1_data, city2_data)
        cat(paste0(pair[1], " vs ", pair[2], ": p-value = ", 
                   round(wilcox_test$p.value, 4), "\n"))
      }
    }
  }
}
